SELECT 
    row_number() OVER (ORDER BY t.trans_time DESC) AS "STT 序号",
    TO_CHAR(t.trans_time, 'YYYY-MM-DD HH24:MI:SS') AS "Thời gian 时间",
    t.sn_code AS "Mã vật liệu 物料编号（新）",
    p.name_vn AS "Tên vật liệu 物料名称",
    p.specifications AS "Quy cách 规格型号",
    t.quantity AS "Số lượng 数量",
    p.unit AS "Đơn vị 单位",
    t.emp_code AS "Mã nhân viên 员工代码",
    p.current_qty AS "Số lượng hiện tại 剩余数量",
    t.urgency_level AS "Mức độ khẩn cấp 紧急程度",
    t.notes AS "Ghi chú 备注"
FROM transactions t
LEFT JOIN parts_catalog p ON t.sn_code = p.part_code
WHERE t.trans_type = 'NHAP_KHO'
ORDER BY t.trans_time DESC;
