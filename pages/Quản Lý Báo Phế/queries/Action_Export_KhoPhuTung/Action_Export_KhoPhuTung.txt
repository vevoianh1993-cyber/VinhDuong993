WITH cap_nhat_kho AS (
    -- Bước 1: Trừ kho và trả về mã vật liệu nếu trừ thành công
    UPDATE parts_catalog 
    SET current_qty = current_qty - {{ this.params.so_luong }},
        actual_qty = actual_qty - {{ this.params.so_luong }}
    WHERE part_code = {{ this.params.ma }}
    RETURNING part_code
)
-- Bước 2: Bắt lấy mã vừa trừ thành công để ghi lịch sử (Chạy trong cùng 1 nhịp)
INSERT INTO scrap_transactions (trans_time, trans_type, item_code, item_type, quantity, notes, emp_code)
SELECT NOW(), 'EXPORT', part_code, 'PART', {{ this.params.so_luong }}, 'Xuất sang báo phế', 'SYSTEM'
FROM cap_nhat_kho;
