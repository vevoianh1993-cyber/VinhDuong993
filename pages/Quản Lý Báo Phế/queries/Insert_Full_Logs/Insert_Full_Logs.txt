-- Dùng phương pháp kiểm tra: Cập nhật cộng dồn nếu đã có (và đang Pending)
WITH updated AS (
    UPDATE scrap_management 
    SET quantity = quantity + {{this.params.qty}}
    WHERE tool_code = {{this.params.code}} 
      AND status = 'Pending'
    RETURNING id
)
-- Nếu không có dòng nào được cộng dồn (tức là mã mới), thì mới Thêm mới
INSERT INTO scrap_management (
    workshop, tool_code, tool_name, unit, quantity, requester, 
    material_type, usage_location, scrap_reason, approver, 
    status, created_at, scrap_status, notes
) 
SELECT 
    {{this.params.xuong}}, {{this.params.code}}, {{this.params.name}}, {{this.params.unit}}, {{this.params.qty}}, {{this.params.emp}}, 
    {{this.params.loai}}, {{this.params.vitri}}, {{this.params.nguyennhan}}, {{this.params.nguoiduyet}}, 
    'Pending', NOW(), {{this.params.trangthai}}, {{this.params.ghichu}}
WHERE NOT EXISTS (SELECT 1 FROM updated);
