WITH stats AS (
    SELECT 
        -- 1. Tổng tồn kho hiện tại
        (SELECT COALESCE(SUM(current_qty), 0) FROM parts_catalog) AS ton_kho,
        
        -- 2. Tổng xuất dùng trong tháng
        (SELECT COALESCE(SUM(quantity), 0) FROM transactions 
         WHERE trans_type IN ('XUAT_KHO', 'EXPORT') 
         AND trans_time::timestamp >= date_trunc('month', CURRENT_DATE)) AS xuat_dung,
         
        -- 3. Tổng báo phế trong tháng (Gom từ 2 bảng cho chắc ăn)
        (
            COALESCE((SELECT SUM(quantity) FROM transactions WHERE trans_type = 'XUAT_PHE' AND trans_time::timestamp >= date_trunc('month', CURRENT_DATE)), 0) +
            COALESCE((SELECT SUM(quantity) FROM scrap_transactions WHERE trans_time::timestamp >= date_trunc('month', CURRENT_DATE)), 0)
        ) AS bao_phe
)
SELECT 
    ton_kho, xuat_dung, bao_phe,
    (ton_kho + xuat_dung + bao_phe) AS tong_tai_san,
    ROUND((xuat_dung::numeric / NULLIF(ton_kho + xuat_dung + bao_phe, 0)) * 100, 1) AS ty_le_su_dung,
    ROUND((bao_phe::numeric / NULLIF(ton_kho + xuat_dung + bao_phe, 0)) * 100, 1) AS ty_le_bao_phe,
    ROUND((ton_kho::numeric / NULLIF(ton_kho + xuat_dung + bao_phe, 0)) * 100, 1) AS ty_le_ton_kho
FROM stats;
